export async function cleanupStaleMiroBoardsJob(): Promise<void> {
  const delay = (ms) => new Promise((resolve) => setTimeout(resolve, ms));

  try {
    // Initialize offset and limit for batch querying
    let offset = 20000;
    const limit = 50;

    // Initialize date from five days ago
    const fiveDaysAgo = new Date();
    fiveDaysAgo.setDate(fiveDaysAgo.getDate() - 5);
    const serviceUserId = process.env.SERVICE_USER_ID;

    // Initialize next link
    let nextLink = `https://api.miro.com/v2/boards?limit=${limit}&offset=${offset}`;

    // While there's a next link
    while (nextLink) {
      // Query boards with current offset
      const response = await axiosWrapper.get(nextLink);
      const boards = response.data;

      for (const board of boards.data) {
        const lastModifiedDate = new Date(board.modifiedAt);

        // If the board has not been modified in the last five days by a user other than the service user, delete it
        if (
          lastModifiedDate < fiveDaysAgo &&
          board.modifiedBy.id === serviceUserId

            `https://api.miro.com/v2/boards/${board.id}`
          );
        }
      }

      await delay(60000);
